<!DOCTYPE html>
<html>

    <title>Form Validation</title>
    <style>
        body { font-family: Arial; background: #f2f2f2; }

        function validateForm() {
            let name = document.getElementById("name").value;
            let roll = document.getElementById("roll").value;
            let mobile = document.getElementById("mobile").value;
            let email = document.getElementById("email").value;

            let namePattern = /^[A-Za-z ]+$/;
            let rollPattern = /^[0-9]+$/;
            let mobilePattern = /^[0-9]{10}$/;

            if (name == "" || !namePattern.test(name)) { alert("Enter valid name (only characters)"); return false; }
            if (roll == "" || !rollPattern.test(roll)) { alert("Roll number must contain only digits"); return false; }
            if (!mobilePattern.test(mobile)) { alert("Mobile number must be 10 digits"); return false; }
            if (!(email.includes("@") && email.includes("."))) { alert("Enter valid email"); return false; }

            alert("Form submitted successfully!");
            return true;
        }
    </script>
</head>

11. ReactJS API Fetch using AXIOS
App.jsx

import { useState, useEffect } from 'react';
import axios from 'axios';

function App() {
  const [userData, setUserData] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const fetchData = async () => {
      try {
        const response = await axios.get('https://jsonplaceholder.typicode.com/users');
        setUserData(response.data);
        setLoading(false);
      } catch (err) {
        setError(err.message);
        setLoading(false);
      }
    };
    fetchData(); 
  }, []);

  if (loading) return <h2 style={{ textAlign: 'center', marginTop: '50px' }}>Loading API Data...</h2>;
  if (error) return <h2 style={{ color: 'red', textAlign: 'center' }}>Error: {error}</h2>;

  return (
    <div style={{ padding: '20px', fontFamily: 'sans-serif', maxWidth: '600px', margin: '0 auto' }}>
      <h1 style={{ textAlign: 'center', borderBottom: '2px solid #333', paddingBottom: '10px' }}>
        Public Directory
      </h1>
      <div style={{ marginTop: '20px' }}>
        {userData.map((user) => (
          <div key={user.id} style={{ border: '1px solid #ccc', borderRadius: '8px', padding: '15px', marginBottom: '15px', backgroundColor: '#f9f9f9' }}>
            <h3 style={{ margin: '0 0 10px 0', color: '#0056b3' }}>{user.name}</h3>
            <p style={{ margin: '5px 0' }}><strong>Email:</strong> {user.email}</p>
            <p style={{ margin: '5px 0' }}><strong>City:</strong> {user.address.city}</p>
          </div>
        ))}
      </div>
    </div>
  );
}
export default App;

import { useState } from 'react';

function App() {
  const [count, setCount] = useState(0);

  const increment = () => { setCount(count + 1); };
  const decrement = () => { setCount(count - 1); };

  return (
    <div style={{ textAlign: 'center', marginTop: '50px', fontFamily: 'system-ui' }}>
      <h1>Interactive Counter</h1>
      
      <h2 style={{ fontSize: '64px', margin: '20px 0', color: '#333' }}>{count}</h2>

      <div style={{ display: 'flex', justifyContent: 'center', gap: '15px' }}>

        table { border-collapse: collapse; width: 100%; max-width: 800px; margin-bottom: 40px; background-color: white; box-shadow: 0 2px 5px rgba(0,0,0,0.1); }
        th, td { border: 1px solid #ddd; padding: 12px; text-align: left; }
        th { background-color: #007BFF; color: white; }
    </style>
</head>
<body>
    <h1>Hospital Management Dashboard</h1>
    <h2>Doctors Directory</h2>
    <table id="doctorsTable">
        <tr><th>ID</th><th>Name</th><th>Specialty</th></tr>
    </table>
    <h2>Patient Records</h2>
    <table id="patientsTable">
        <tr><th>ID</th><th>Name</th><th>Age</th><th>Ailment</th></tr>
    </table>

    <script>
        const xmlString = `<?xml version="1.0" encoding="UTF-8"?>
        <Hospital>
            <Doctors>
                <Doctor id="D01">
                    <Name>Dr. Alice Smith</Name>
                    <Specialty>Cardiology</Specialty>
                </Doctor>
                <Doctor id="D02">
                    <Name>Dr. Bob Jones</Name>
                    <Specialty>Neurology</Specialty>
                </Doctor>
            </Doctors>
            <Patients>
                <Patient id="P01">
                    <Name>John Doe</Name>
                    <Age>45</Age>
                    <Ailment>Hypertension</Ailment>
                </Patient>
                <Patient id="P02">
                    <Name>Jane Roe</Name>
                    <Age>32</Age>
                    <Ailment>Migraine</Ailment>
                </Patient>
            </Patients>
        </Hospital>`;

        const parser = new DOMParser();
        const xmlDoc = parser.parseFromString(xmlString, "text/xml");

        function loadDoctors() {
            const doctors = xmlDoc.getElementsByTagName("Doctor");
            const table = document.getElementById("doctorsTable");
            for (let i = 0; i < doctors.length; i++) {
                let row = table.insertRow(-1);
                let id = doctors[i].getAttribute("id");
                let name = doctors[i].getElementsByTagName("Name")[0].textContent;
                let specialty = doctors[i].getElementsByTagName("Specialty")[0].textContent;
                row.insertCell(0).innerHTML = id;
                row.insertCell(1).innerHTML = name;
                row.insertCell(2).innerHTML = specialty;
            }
        }

        function loadPatients() {
            const patients = xmlDoc.getElementsByTagName("Patient");
            const table = document.getElementById("patientsTable");
            for (let i = 0; i < patients.length; i++) {
                let row = table.insertRow(-1);
                let id = patients[i].getAttribute("id");
                let name = patients[i].getElementsByTagName("Name")[0].textContent;
                let age = patients[i].getElementsByTagName("Age")[0].textContent;
                let ailment = patients[i].getElementsByTagName("Ailment")[0].textContent;
                row.insertCell(0).innerHTML = id;
                row.insertCell(1).innerHTML = name;
                row.insertCell(2).innerHTML = age;
                row.insertCell(3).innerHTML = ailment;
            }
        }
        loadDoctors();
        loadPatients();
    </script>
</body>
</html>
