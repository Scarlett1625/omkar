            let name = document.getElementById("name").value;
            let roll = document.getElementById("roll").value;
            let mobile = document.getElementById("mobile").value;
            let email = document.getElementById("email").value;

            let namePattern = /^[A-Za-z ]+$/;
            let rollPattern = /^[0-9]+$/;
            let mobilePattern = /^[0-9]{10}$/;

            // Name validation
            if (name == "" || !namePattern.test(name)) {
                alert("Enter valid name (only characters)");
                return false;
            }

            // Roll number validation
            if (roll == "" || !rollPattern.test(roll)) {
                alert("Roll number must contain only digits");
                return false;
            }

            // Mobile validation
            if (!mobilePattern.test(mobile)) {
                alert("Mobile number must be 10 digits");
                return false;
            }

            // Email validation
            if (!(email.includes("@") && email.includes("."))) {
                alert("Enter valid email");
                return false;
            }

            alert("Form submitted successfully!");
            return true;
        }

    $cookie_value = $_POST['uname'];

    echo "<p>Cookie updated successfully!</p>";
}

// Delete Cookie
if (isset($_POST['delete'])) {
    setcookie($cookie_name, "", time() - 3600, "/");

import { useState, useEffect } from 'react';
import axios from 'axios';

function App() {
  const [userData, setUserData] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const fetchData = async () => {
      try {
        const response = await axios.get('https://jsonplaceholder.typicode.com/users');
        setUserData(response.data);
        setLoading(false);
      } catch (err) {
        setError(err.message);
        setLoading(false);
      }
    };

    fetchData(); 
  }, []);
  if (loading) return <h2 style={{ textAlign: 'center', marginTop: '50px' }}>Loading API Data...</h2>;
  if (error) return <h2 style={{ color: 'red', textAlign: 'center' }}>Error: {error}</h2>;
  return (
    <div style={{ padding: '20px', fontFamily: 'sans-serif', maxWidth: '600px', margin: '0 auto' }}>
      <h1 style={{ textAlign: 'center', borderBottom: '2px solid #333', paddingBottom: '10px' }}>
        Public Directory
      </h1>
      
      <div style={{ marginTop: '20px' }}>
        {userData.map((user) => (
          <div key={user.id} style={{ 
            border: '1px solid #ccc', 
            borderRadius: '8px', 
            padding: '15px', 
            marginBottom: '15px',
            backgroundColor: '#f9f9f9'
          }}>
            <h3 style={{ margin: '0 0 10px 0', color: '#0056b3' }}>{user.name}</h3>
            <p style={{ margin: '5px 0' }}><strong>Email:</strong> {user.email}</p>
            <p style={{ margin: '5px 0' }}><strong>City:</strong> {user.address.city}</p>
          </div>
        ))}
      </div>
    </div>
  );
}

export default App;

import { useState } from 'react';
function App() {
  // 1. Set up a state variable to track which message to show.
  // We initialize it to 'true' so the default is the Welcome message.
  const [isWelcome, setIsWelcome] = useState(true);
  // 2. Create a function to handle the button click.
  // This flips the state: if true it becomes false, if false it becomes true.
  const handleToggle = () => {
    setIsWelcome(!isWelcome);
  };
  return (
    <div style={{ textAlign: 'center', marginTop: '50px', fontFamily: 'system-ui' }}>
      
      {/* 3. Conditional Rendering:
          We use a ternary operator (condition ? true_result : false_result) 
          to decide which text to display on the screen. */}
      <h1>
        {isWelcome ? "Welcome Student" : "Thank You"}
      </h1>

      {/* 4. The Button: triggers the handleToggle function when clicked */}
      <button 
        onClick={handleToggle}
        style={{
          padding: '10px 20px',
          fontSize: '16px',
          cursor: 'pointer',
          marginTop: '20px',
          backgroundColor: '#007BFF',
          color: 'white',
          border: 'none',
          borderRadius: '5px'
        }}
      >
        Toggle Message
      </button>

    </div>
  );
}
